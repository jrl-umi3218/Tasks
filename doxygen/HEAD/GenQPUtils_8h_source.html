30 inline void fillQC(const std::vector<Task *> & tasks, int nrVars, Eigen::MatrixXd & Q, Eigen::VectorXd & C)
31 {
32  for(std::size_t i = 0; i < tasks.size(); ++i)
33  {
34  const Eigen::MatrixXd & Qi = tasks[i]->Q();
35  const Eigen::VectorXd & Ci = tasks[i]->C();
36  std::pair<int, int> b = tasks[i]->begin();
37 
38  int r = static_cast<int>(Qi.rows());
39  int c = static_cast<int>(Qi.cols());
40 
41  Q.block(b.first, b.second, r, c) += tasks[i]->weight() * Qi;
42  C.segment(b.first, r) += tasks[i]->weight() * Ci;
43  }
44 
45  // try to transform Q_ to a positive matrix
46  // we just add a small value to the diagonal since
47  // the first necessary condition is to have
48  // Q_(i,i) > 0
49  // may be we can try to check the second
50  // condition in a near futur
51  // Q_(i,i) + Q_(j,j) > 2·Q_(i,j) for i≠j
52  for(int i = 0; i < nrVars; ++i)
53  {
54  if(std::abs(Q(i, i)) < DIAG_CONSTANT)
55  {
56  Q(i, i) += DIAG_CONSTANT;
57  }
58  }
59 }

64 inline void reduceQC(const Eigen::MatrixXd & QFull,
65  const Eigen::VectorXd & CFull,
66  Eigen::MatrixXd & Q,
67  Eigen::VectorXd & C,
68  const std::vector<int> & fullToReduced,
69  const std::vector<int> & reducedToFull,
70  const std::vector<std::tuple<int, int, double>> & dependencies)
71 {
72  /* Start by moving the non-reduced variables to their new location */
73  for(size_t i = 0; i < reducedToFull.size(); ++i)
74  {
75  for(size_t j = i; j < reducedToFull.size(); ++j)
76  {
77  Q(i, j) = QFull(reducedToFull[i], reducedToFull[j]);
78  }
79  C(i) = CFull(reducedToFull[i]);
80  }
81  /* Apply the reduction */
82  for(const auto & d : dependencies)
83  {
84  const int & primaryFullI = std::get<0>(d);
85  const int & primaryReducedI = fullToReduced[primaryFullI];
86  const int & replicaFullI = std::get<1>(d);
87  const double & alpha = std::get<2>(d);
88  /* Apply reduction to C */
89  C(primaryReducedI) += alpha * CFull(replicaFullI);
90  /* Add cross-terms to Q */
91  for(size_t i = 0; i < reducedToFull.size(); ++i)
92  {
93  Q(i, primaryReducedI) += alpha * QFull(reducedToFull[i], replicaFullI);
94  }
95  /* Add diagonal element to Q */
96  Q(primaryReducedI, primaryReducedI) += alpha * alpha * QFull(replicaFullI, replicaFullI);
97  }
98  /* Update the lower triangular part of Q */
99  Q = Q.selfadjointView<Eigen::Upper>();
100 }

108 inline int fillEq(const std::vector<Equality *> & eq,
109  int nrVars,
110  int nrALines,

124  AL.segment(nrALines, nrConstr) = bi.head(nrConstr);
125  AU.segment(nrALines, nrConstr) = bi.head(nrConstr);

127  nrALines += nrConstr;

298  Eigen::MatrixXd & A,

321  Eigen::VectorXd & XL,
322  const Eigen::VectorXd & XUFull,
323  Eigen::VectorXd & XU,

336  const int & primaryReducedI = fullToReduced[primaryFullI];
337  const int & replicaFullI = std::get<1>(d);
338  const double & alpha = std::get<2>(d);
339  if(alpha != 0)
340  {
342  if(alpha < 0)
343  {
344  XL(primaryReducedI) = std::max(XL(primaryReducedI), XUFull(replicaFullI) / alpha);
345  XU(primaryReducedI) = std::min(XU(primaryReducedI), XLFull(replicaFullI) / alpha);
346  }
347  else
348  {
349  XL(primaryReducedI) = std::max(XL(primaryReducedI), XLFull(replicaFullI) / alpha);
350  XU(primaryReducedI) = std::min(XU(primaryReducedI), XUFull(replicaFullI) / alpha);
351  }

359 inline void expandResult(const Eigen::VectorXd & result,
360  Eigen::VectorXd & resultFull,
361  const std::vector<int> & reducedToFull,
362  const std::vector<std::tuple<int, int, double>> & dependencies)
363 {
364  for(size_t i = 0; i < reducedToFull.size(); ++i)
365  {
366  resultFull(reducedToFull[i]) = result(i);

371  const int & replicaFullI = std::get<1>(d);
